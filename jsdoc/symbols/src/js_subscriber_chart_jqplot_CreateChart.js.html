1 /**
  2  * to report errors related to addScript in gadgets
  3  * @param errString the error string
  4  */
  5 Viskit.s.chart.jqplot.reportError =  function (errString) {
  6     if(!window.__error) window.__error = {};
  7     if( !window.__error[errString] ) {
  8         // Suppress multiple
  9         window.__error[errString] = true;
 10         alert(errString);
 11     }
 12 };
 13 /**
 14  *  @return returns the module base
 15  */
 16 Viskit.s.chart.jqplot.getModuleBase = function () {
 17     if( window.__moduleBase ) return window.__moduleBase;
 18     if( _args ) {
 19         var moduleBase=_args()['url'];
 20         moduleBase=moduleBase.substring(0,moduleBase.lastIndexOf('/')+1);
 21         window.__moduleBase = moduleBase;
 22         return window.__moduleBase;
 23     };
 24     this.reportError('Can not find module base. Gadget may not work properly.');
 25     return '';
 26 };
 27 /**
 28  * returns the rebase relative URL
 29  * @param relativeUrl the relative URL
 30  * @param cached
 31  * @return returns the rebase relative URL
 32  */
 33 Viskit.s.chart.jqplot.rebaseRelativeUrl = function (relativeUrl,cached) {
 34     var moduleBase = this.getModuleBase();
 35     var absUrl = moduleBase+relativeUrl;
 36     if( cached && _IG_GetCachedUrl ) {
 37         absUrl = _IG_GetCachedUrl(absUrl);
 38     };
 39     return absUrl;
 40 };
 41 /**
 42  * adds the css given by the URL into the gadget css
 43  * @param cssRelativeUrl the relative URL of the CSS sheet
 44  */
 45 Viskit.s.chart.jqplot.addStylesheet = function (cssRelativeUrl) {
 46     var rebasedUrl = this.rebaseRelativeUrl(cssRelativeUrl,true);
 47     document.write('<link rel="stylesheet" href="'+rebasedUrl+'">')
 48 };

 50 /**
 51  * adds the script file given by the url into the gadget
 52  * @param jsRelativeUrl the relative URL of the script file
 53  */
 54 Viskit.s.chart.jqplot.addScript = function (jsRelativeUrl,callback) {
 55     var rebasedUrl = this.rebaseRelativeUrl(jsRelativeUrl,true);
 56     document.write('<script type="text/javascript" src="'+rebasedUrl+'"><\/script>')
 57     if(callback){
 58 		callback();
 59 	}
 60 };
