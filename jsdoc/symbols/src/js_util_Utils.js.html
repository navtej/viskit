 20 /**
 21  * find the max value of a given array
 22  *
 23  * @returns the max value of the array
 24  */
 25 Array.prototype.max = function() {
 26     var max = this[0];
 27     var len = this.length;
 28     for (var i = 1; i < len; i++) if (this[i] > max) max = this[i];
 29     return max;
 30 }
 31 /**
 32  * find the min value of a given array
 33  * @return the min value of the array
 34  */
 35 Array.prototype.min = function() {
 36     var min = this[0];
 37     var len = this.length;
 38     for (var i = 1; i < len; i++) if (this[i] < min) min = this[i];
 39     return min;
 40 }
 41 /**
 42  *
 43  * @param count
 44  * @param scheme
 45  */
 46 Viskit.util.generateColors = function(count, scheme) {
 47     function hexNumtoHexStr(n) {
 48         function toHexStr(N) {
 49              if (N==null) return "00";
 50              N=parseInt(N); if (N==0 || isNaN(N)) return "00";
 51              N=Math.max(0,N); N=Math.min(N,255); N=Math.round(N);
 52              return "0123456789ABCDEF".charAt((N-N%16)/16)
 53                   + "0123456789ABCDEF".charAt(N%16);
 54         }
 55         
 56         return "#" + toHexStr((n & 0xFF0000)>>>16) + toHexStr((n & 0x00FF00)>>>8) + toHexStr((n & 0x0000FF));
 57     }
 58         
 59     function generateInterpolatedColorArray(count, colors) {
 60         function interpolateColors(color1, color2, f) {
 61             if (f >= 1)
 62                 return color2;
 63             if (f <= 0)
 64                 return color1;
 65             var fb = 1 - f;
 66             return ((((color2 & 0xFF0000) * f)+((color1 & 0xFF0000) * fb)) & 0xFF0000)
 67                    +((((color2 & 0x00FF00) * f)+((color1 & 0x00FF00) * fb)) & 0x00FF00)
 68                    +((((color2 & 0x0000FF) * f)+((color1 & 0x0000FF) * fb)) & 0x0000FF);                                   
 69         }               
 70         
 71         var len = colors.length;
 72         var res = new Array();
 73         res.push(hexNumtoHexStr(colors[0]));
 74         
 75         for (var i = 1; i < count; i++) {
 76             var val = i * len / count;
 77             var color1 = Math.floor(val);
 78             var color2 = Math.ceil(val);
 79             res.push(hexNumtoHexStr(interpolateColors(colors[color1], colors[color2], val - color1)));                        
 80         }
 81         
 82         return res;
 83     }
 84 
 85     if (count <= 0) 
 86         return null;
 87         
 88     var a10 = [0x1f77b4, 0xff7f0e, 0x2ca02c, 0xd62728, 0x9467bd,
 89         0x8c564b, 0xe377c2, 0x7f7f7f, 0xbcbd22, 0x17becf];
 90     var b20 = [0x1f77b4, 0xaec7e8, 0xff7f0e, 0xffbb78, 0x2ca02c,
 91           0x98df8a, 0xd62728, 0xff9896, 0x9467bd, 0xc5b0d5,
 92           0x8c564b, 0xc49c94, 0xe377c2, 0xf7b6d2, 0x7f7f7f,
 93           0xc7c7c7, 0xbcbd22, 0xdbdb8d, 0x17becf, 0x9edae5];
 94     var c19 = [0x9c9ede, 0x7375b5, 0x4a5584, 0xcedb9c, 0xb5cf6b,
 95           0x8ca252, 0x637939, 0xe7cb94, 0xe7ba52, 0xbd9e39,
 96           0x8c6d31, 0xe7969c, 0xd6616b, 0xad494a, 0x843c39,
 97           0xde9ed6, 0xce6dbd, 0xa55194, 0x7b4173];
 98     var colorScheme;
 99     
100     if (scheme == 20) {
101         colorScheme = b20;
102     }
103     else if (scheme == 10) {
104         colorScheme = a10;
105     }
106     else /* any ((scheme === undefined) || (scheme == 19))*/{
107         colorScheme = c19;
108     }
109     
110     if (count <= colorScheme.length) {
111         c = new Array();
112         for (var i = 0; i < count; i++)
113             c.push(hexNumtoHexStr(colorScheme[i]));
114         return c;
115     }
116     
117     return generateInterpolatedColorArray(count, colorScheme);
118 }
