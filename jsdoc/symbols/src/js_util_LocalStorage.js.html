 13 Viskit.util.genaratekey = function(value){
 14 	var path = document.location.href;
 15 	
 16 	var startindex = 0;
 17 	var endindex = 0;
 18 		if(path.indexOf('http://') !== -1){
 19 			startindex = path.indexOf('http://') + 7;
 20 		}else if(path.indexOf('https://') != -1){
 21 			startindex = path.indexOf('http://') + 8;
 22 		}
 23 		
 24 		if(path.lastIndexOf('.') > path.lastIndexOf('/')){
 25 			endindex = path.lastIndexOf('.') +1;
 26 		}
 27 		if(endindex == 0){
 28 			path = path.substring(startindex);
 29 		}else{
 30 			path = path.substring(startindex,endindex);
 31 		}
 32 		path = path.replace(new RegExp('/', 'g'),'.')
 33 		
 34 		if((path.lastIndexOf('.')+1) === path.length){
 35 			path = path + value;
 36 		}else{
 37 			path = path + '.' + value;	
 38 		}
 39 		
 40 	return path;
 41 }

 97 /**
 98  * set an item into the local storage with the given value and key
 99  * if an item already exists for that key the value will be overridden
100  * @param key the key that is used to store the data in the local storage
101  * @param value  the string value to be stored
102  * @return returns true if there was no value by the key and false if the key exists
103  */
104 Viskit.util.localStorage.setItem = function(key,value){
105 	if(localStorage.getItem(key) != null){
106 		Viskit.util.localStorage.clear(key);
107 		localStorage.setItem(key,value);
108 		return false;
109 	} else{
110 		localStorage.setItem(key,value);
111 		return true;
112 	}
113 		
114 }
