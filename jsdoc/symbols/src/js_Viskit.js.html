240 Viskit.fn.traverseToDataField = function (object, dataFieldArray) {
241 	var a = object;					
242 	for (var i = 0; i < dataFieldArray.length; i++) {
243 		a = a[dataFieldArray[i]];
244 	}
245 	return a;
246 };

248 Viskit.fn.traverseNKillLeaf = function (object, dataFieldArray) {
249 	var a = object;
250 	for (var i = 0; i < dataFieldArray.length; i++) {
251 		if (i == dataFieldArray.length - 1) {
252 			delete a[dataFieldArray[i]];
253 		}
254 		else {
255 			a = a[dataFieldArray[i]];
256 		}
257 	}
258 };
