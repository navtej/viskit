 73 Viskit.s.chart.Chart.prototype.pushData = function (d) {
 74     if (this.validateData(d)) {
 75         this.data = d;
 76         this.preUpdate();
 77         this.update();
 78         this.postUpdate();
 79     } else {
 80         this.updateMessageDiv(this.messageInterceptFunction());
 81     }
 82 };

 84 Viskit.s.chart.Chart.prototype.validateData = function (d) {
 85     //Check whether we have valid data or not.
 86     if (d === null || d === undefined) {
 87         return false;
 88     }
 89     else {
 90         return true;
 91     }
 92 };

 97 Viskit.s.chart.Chart.prototype.updateMessageDiv = function (s) {
 98 
 99     if (this.msgDiv() !== undefined) {
100         var msgdiv = document.getElementById(this.msgDiv());
101         if (msgdiv !== undefined) {
102             msgdiv.innerHTML = s;
103             msgdiv.style.display = "block";
104         }
105     }
106 };

141 Viskit.s.chart.Chart.prototype.traverseToDataField = function (object, dataFieldArray) {
142     var a = object;
143     try { //Try catch outside the loop TODO
144         for (var i = 0; i < dataFieldArray.length; i++) {
145             a = a[dataFieldArray[i]];
146         }
147     }
148     catch (e) {
149         this.updateMessageDiv(this.messageInterceptFunction());
150     }
151     return a;
152 };
