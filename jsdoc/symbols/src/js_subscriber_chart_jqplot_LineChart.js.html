  8 /**
  9  *
 10  * Class jqplot.LineChart : Chart
 11  * This is the custom wrapper class for jqplot line charts
 12  * creates a line chart with the data provided
 13  * @param canvas the div to which the chart will be plotted
 14  * @param chartTitle the Title of the chart
 15  * @param chartDesc
 16  */
 17 Viskit.s.chart.jqplot.LineChart = function(canvas, chartTitle, chartDesc) {
 18     Viskit.s.chart.jqplot.Chart.call(this, canvas, chartTitle, chartDesc);
 19     this.canvas = canvas;
 20     this.heightratio = 1;
 21     this.widthratio = 1;
 22     this.dataHistory = [];
 23     this.maxX = 0;
 24     this.maxY = 0;
 25     this.fillchart(false)
 26     .showPointLabels(false)
 27     .seriesLabels([])
 28     .showMarker(false)
 29     .showToolTip(true)
 30     .tooltipFollowMouse(false)
 31     .enableZoom(false)
 32     .xAxisType('time')
 33 	.titleTop(chartTitle)
 34     .yAxisType('linear');
 35 	 
 36 }

356 Viskit.s.chart.jqplot.LineChart.prototype.populateData = function (thisObject) {
357 	
358     var _dataField = thisObject.traverseToDataField(thisObject.data, thisObject.dataField());
359     this.x =true;
360     var dataGrpCount = 1;
361     if( _dataField instanceof Array ) {
362         dataGrpCount = _dataField.length;
363 		
364     }
365 
366     thisObject.formattedData = [];
367     if(thisObject.enableLocalStorage()){
368         if(Viskit.util.localStorage.exsists(Viskit.util.genaratekey("jqplot."+thisObject.divEl()))){
369 		
370             Viskit.util.localStorage.manageStore(thisObject.storageTime(),Viskit.util.genaratekey("jqplot."+thisObject.divEl()));
371             var jsonString = Viskit.util.localStorage.getItem(Viskit.util.genaratekey("jqplot."+thisObject.divEl()));
372             thisObject.dataHistory = JSON.parse(jsonString);
373 			
374         }
375     }else if(thisObject.enableSessionStorage()){
376         if(Viskit.util.sessionStorage.exsists(Viskit.util.genaratekey("jqplot."+thisObject.divEl()))){
377             Viskit.util.sessionStorage.manageStore(thisObject.storageTime(),Viskit.util.genaratekey("jqplot."+thisObject.divEl()));
378             var jsonString = Viskit.util.sessionStorage.getItem(Viskit.util.genaratekey("jqplot."+thisObject.divEl()));
379             thisObject.dataHistory = JSON.parse(jsonString);
380 			
381         }
382 		
383     }
384     for(var i=0;i<dataGrpCount;i++){
385 		
386         thisObject.formattedData.push(genDataMap(i))
387     }
388     if(thisObject.enableLocalStorage()){
389 		
390         var jsonData = JSON.stringify(thisObject.dataHistory);
391         Viskit.util.localStorage.setItem(Viskit.util.genaratekey("jqplot."+thisObject.divEl()),jsonData);
392     }else if(thisObject.enableSessionStorage()){
393         var jsonData = JSON.stringify(thisObject.dataHistory);
394         Viskit.util.sessionStorage.setItem(Viskit.util.genaratekey("jqplot."+thisObject.divEl()),jsonData);
395 		
396     }
397   
398     function genDataMap(x) {
399         var rootObj;
400         if( _dataField instanceof Array ) {
401             rootObj = _dataField[x];
402         }
403         else {
404             rootObj = _dataField;
405         }
406         var prevalObj = (thisObject.traverseToDataField(rootObj, thisObject.dataValue()));
407         var valObj = [];
408         if((typeof(thisObject.xAxisType()) == 'undefined')){
409             valObj = prevalObj;
410         }else if(thisObject.xAxisType() === 'time'){
411             valObj = [new Date().getTime(),prevalObj]
412         }
413 		  
414         if( (parseInt(prevalObj) > thisObject.maxY)){
415             thisObject.maxY = parseInt(prevalObj);
416 			
417         }
418 		
419         Viskit.util.localStorage.clearAll();
420         if(thisObject.dataHistory[x] === undefined){
421             thisObject.dataHistory[x] = new Array();
422 						
423         }
424         if (thisObject.dirFromLeft()) {
425             thisObject.dataHistory[x].unshift(valObj);
426 
427             if(thisObject.dataHistory[x].length > thisObject.band()+1){
428                 thisObject.dataHistory[x].pop();
429             }
430         }
431         else {
432             thisObject.dataHistory[x].push(valObj);
433 
434             if(thisObject.dataHistory[x].length > thisObject.band()+1){
435                 thisObject.dataHistory[x].shift();
436             }
437         }
438 		
439         return thisObject.dataHistory[x];
440     }
441     this.setOptions(thisObject);
442 
443 };

450 Viskit.s.chart.jqplot.LineChart.prototype.update = function () {
451     this.populateData(this);
452     document.getElementById(this.divEl()).innerHTML = ""
453     plot2 = $.jqplot(this.canvas,this.getData(this),this.options);
454     plot2.replot();
455     heightratio = this.heightratio;
456     widthratio = this.widthratio;
457     var chartid = '#' + this.divEl();
458     $(function(){
459 	
460         $(window).resize(function(){
461 	
462             var h = $(window).height()*heightratio;
463             var w = $(window).width()*widthratio;
464 		
465             if( h > 250 ){
466                 $(chartid).height(h);
467             }else{
468                 $(chartid).height(250);
469             }
470 		
471             if( w > 350 ){
472                 $(chartid).width(w);
473             }else{
474                 $(chartid).width(350);
475             }
476 		
477         
478 		
479             plot2.replot();
480         });
481     });
482     if(this.tooltip() === true) {
483         tooltip.init();
484     }
485 };

490 Viskit.s.chart.jqplot.LineChart.prototype.getDataLabel = function (i) {
491     if (this.data !== null){
492         var rootObj = this.traverseToDataField(this.data, this.dataField());
493         if( rootObj instanceof Array ) {
494             return  this.traverseToDataField(rootObj[i], this.dataLabel());
495         }
496         else {
497             return  this.traverseToDataField(rootObj, this.dataLabel());
498         }
499     }
500     return i;
501 };
