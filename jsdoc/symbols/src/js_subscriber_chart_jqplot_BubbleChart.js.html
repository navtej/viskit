  8 /**
  9  * Class jqplot.Bubble : Chart
 10  * This is the custom wrapper class for jqplot bubble charts
 11  * creates a bubble chart with the data provided
 12  * @param canvas the div to which the chart will be plotted
 13  * @param chartTitle the Title of the chart
 14  * @param chartDesc

 16 Viskit.s.chart.jqplot.BubbleChart = function(canvas, chartTitle, chartDesc) {
 17     Viskit.s.chart.jqplot.Chart.call(this, canvas, chartTitle, chartDesc);
 18     this.canvas = canvas;
 19     this.heightratio = 1;
 20     this.widthratio = 1;
 21     this.varyBubbleColors(true)
 22     .autoscaleBubbles(true)
 23     .autoscaleMultiplier(1.0)
 24     .autoscalePointsFactor(-0.07)
 25     .escapeHtml(true)
 26     .highlightMouseOver(true)
 27     .highlightMouseDown(false)
 28     .highlightColors([])
 29     .bubbleAlpha(1.0)
 30     .highlightAlpha(null)
 31     .bubbleGradients(false)
 32     .showLabels(true)
 33     .showPointLabels(false)
 34     .seriesLabels([])
 35     .showLegend(true)
 36     .dataValue2([])
 37     .dataValue3([])
 38 	.titleTop(chartTitle);

 40   
 41 }

165 Viskit.s.chart.jqplot.BubbleChart.prototype.populateData = function (thisObject) {
166     var _dataField = thisObject.traverseToDataField(thisObject.data, thisObject.dataField());
167 	
168     var dataGrpCount = 1;
169     if( _dataField instanceof Array ) {
170         dataGrpCount = _dataField.length;
171 		
172     }
173 
174     thisObject.formattedData = [];
175     for(var i=0;i<dataGrpCount;i++){
176         thisObject.formattedData.push([genDataMap(1,i),genDataMap(2,i),genDataMap(3,i),this.getDataLabel(i)])
177     }
178   
179     function genDataMap(y,x) {
180         var rootObj;
181         if( _dataField instanceof Array ) {
182             rootObj = _dataField[x];
183         }
184         else {
185             rootObj = _dataField;
186         }
187         if(y == 1){
188             return parseInt(thisObject.traverseToDataField(rootObj, thisObject.dataValue()));
189         }else if( y == 2){
190             return parseInt(thisObject.traverseToDataField(rootObj, thisObject.dataValue2()));
191         }else if(y == 3){
192             return parseInt(thisObject.traverseToDataField(rootObj, thisObject.dataValue3()));
193         }
194     }
195     this.setOptions(thisObject)
196 
197 };

199 Viskit.s.chart.jqplot.BubbleChart.prototype.getData = function (thisObject) {
200 
201     return thisObject.formattedData;
202 };

204 Viskit.s.chart.jqplot.BubbleChart.prototype.update = function () {
205     this.populateData(this);
206     document.getElementById(this.divEl()).innerHTML = ""
207     plot2 = $.jqplot(this.canvas,[this.getData(this)],this.options);
208     heightratio = this.heightratio;
209     widthratio = this.widthratio;
210     var chartid = '#' + this.divEl();
211     $(function(){
212 	
213         $(window).resize(function(){
214 	
215             var h = $(window).height()*heightratio;
216             var w = $(window).width()*widthratio;
217 		
218             if( h > 250 ){
219                 $(chartid).height(h);
220             }else{
221                 $(chartid).height(250);
222             }
223 		
224             if( w > 350 ){
225                 $(chartid).width(w);
226             }else{
227                 $(chartid).width(350);
228             }
229 		
230         
231 		
232             plot2.replot();
233         });
234     });
235     if(this.tooltip() === true) {
236         tooltip.init();
237     }
238 };

240 Viskit.s.chart.jqplot.BubbleChart.prototype.getDataLabel = function (i) {
241     if (this.data !== null){
242 
243         var rootObj = this.traverseToDataField(this.data, this.dataField());
244         if( rootObj instanceof Array ) {
245             return  this.traverseToDataField(rootObj[i], this.dataLabel());
246         }
247         else {
248             return  this.traverseToDataField(rootObj, this.dataLabel());
249         }
250     }
251     
252     return i;
253 };
