1 /**
  2 * JSON provider this handles services that retrieve JSON strings
  3 * @class ProviderGET
  4 * @extends Provider
  5 **/
  6 Viskit.p.ProviderGETJSON = function(url, args, that) {
  7 	Viskit.p.ProviderGET.call(this, url, args, that);
  8 };
  9 
 10 Viskit.extend(Viskit.p.ProviderGETJSON, Viskit.p.ProviderGET);
 11 
 12 /**
 13  * parses the response JSON string to a JSON object
 14  * @param {String} response the JSON string that is received
 15  * @param {Object}[that]
 16  * @return { JSON } returns a the JSON object corresponding to the response string
 17  */
 18 Viskit.p.ProviderGETJSON.prototype.parseResponse = function(response, that) {
 19     return JSON.parse(response);
 20 };
