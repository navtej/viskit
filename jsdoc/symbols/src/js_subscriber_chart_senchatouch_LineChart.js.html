253 Viskit.s.chart.sencha.LineChart.prototype.getDataLabel = function (i) {
254     if (this.data !== null){
255 
256 	var rootObj = this.traverseToDataField(this.data, this.dataField());
257 	if( rootObj instanceof Array ) {
258 	    return  this.traverseToDataField(rootObj[i], this.dataLabel());
259 	}
260 	else {
261 	    return  this.traverseToDataField(rootObj, this.dataLabel());
262 	}
263     }
264 
265     return i;
266 };
267 Viskit.s.chart.sencha.LineChart.prototype.getTicks = function(type){
268     var ticks = [];
269 
270     if(type === 'x'){
271 	if(this.dataLabel().length == 0){
272 	    return this.xTicks();
273 	}else{
274 	    var _dataField = this.traverseToDataField(this.data, this.dataField());
275 
276 	    var dataGrpCount = 1;
277 	    if( _dataField instanceof Array ) {
278 		dataGrpCount = _dataField.length;
279 
280 	    }
281 	    for(var j = 0 ; j <dataGrpCount; j++){
282 		ticks.push(this.getDataLabel(j));
283 	    };
284 	    return ticks.reverse();
285 
286 	}
287 
288 
289     }else if(type === 'y'){
290 	return this.yTicks();
291     }
