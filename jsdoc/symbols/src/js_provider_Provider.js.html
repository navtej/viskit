 20 /**
 21  * add a data receiver. the data receiver add will receive data from the provider
 22  * @param dataReceiver the data receiver
 23  */
 24 Viskit.p.Provider.prototype.addDataReceiver = function(dataReceiver) {
 25     this.drList.push(dataReceiver);
 26 };
 27 /**
 28  * loops all the data receiver specified for this provider and pumps data into them
 29  *
 30  * @param data data to be pushed into the data receivers
 31  */
 32 Viskit.p.Provider.prototype.pushData = function(data) {
 33     // loop all data receivers. Pump data to them.
 34     //console.log(JSON.stringify(data) + this.url);
 35     if (this.postValidate(this, data)) {
 36         for (i = 0; i < this.drList.length; i++) {
 37             (this.drList[i]).pushData(data);
 38         }
 39     }
 40 };
