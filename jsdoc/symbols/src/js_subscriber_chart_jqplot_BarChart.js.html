 11  * This is the custom wrapper class for jqplot bar charts
 12  * creates a bar chart with the data provided
 13  * @param canvas the div to which the chart will be plotted
 14  * @param chartTitle the Title of the chart
 15  * @param chartDesc

 17 Viskit.s.chart.jqplot.BarChart = function(canvas, chartTitle, chartDesc) {
 18    Viskit.s.chart.jqplot.Chart.call(this, canvas, chartTitle, chartDesc);
 19     this.canvas = canvas;
 20     this.heightratio = 1;
 21     this.widthratio = 1;
 22     this.xAxisType('category')
 23     .barPadding(8)
 24     .barMargin(10)

 31     .varyBarColor(false)

 33     .highlightMouseDown(false)
 34     .highlightColors([])

335 Viskit.s.chart.jqplot.BarChart.prototype.populateData = function (thisObject) {
336     var _dataField = thisObject.traverseToDataField(thisObject.data, thisObject.dataField());
337 	
338     var dataGrpCount = 1;
339     if( _dataField instanceof Array ) {
340         dataGrpCount = _dataField.length;
341 		
342     }
343 
344     thisObject.formattedData = [];
345     thisObject.tempdata = [];
346     for(var i=0;i<dataGrpCount;i++){
347         thisObject.tempdata.push(genDataMap(i))
348     }
349     thisObject.formattedData.push(thisObject.tempdata.reverse());
350 	 	
351     function genDataMap(x) {
352         var rootObj;
353         if( _dataField instanceof Array ) {
354             rootObj = _dataField[x];
355         }
356         else {
357             rootObj = _dataField;
358         }
359         return parseInt(thisObject.traverseToDataField(rootObj, thisObject.dataValue()));
360 		
361     }
362     this.setOptions(thisObject);
363 };

367     return thisObject.formattedData;

370 Viskit.s.chart.jqplot.BarChart.prototype.update = function () {
371     this.populateData(this);
372     document.getElementById(this.divEl()).innerHTML = ""
373     plot2 = $.jqplot(this.canvas,this.getData(this),this.options);
374     heightratio = this.heightratio;
375     widthratio = this.widthratio;
376     var chartid = '#' + this.divEl();
377     $(function(){
378 	
379         $(window).resize(function(){
380 	
381             var h = $(window).height()*heightratio;
382             var w = $(window).width()*widthratio;
383 		
384             if( h > 250 ){
385                 $(chartid).height(h);
386             }else{
387                 $(chartid).height(250);
388             }
389 		
390             if( w > 350 ){
391                 $(chartid).width(w);
392             }else{
393                 $(chartid).width(350);
394             }
395 		
396         
397 		
398             plot2.replot();
399         });
400     });
401     if(this.tooltip() === true) {
402         tooltip.init();
403     }
404 };

406 Viskit.s.chart.jqplot.BarChart.prototype.getDataLabel = function (i) {
407     if (this.data !== null){
408 
409         var rootObj = this.traverseToDataField(this.data, this.dataField());
410         if( rootObj instanceof Array ) {
411             return  this.traverseToDataField(rootObj[i], this.dataLabel());
412         }
413         else {
414             return  this.traverseToDataField(rootObj, this.dataLabel());
415         }
416     }
417     
418     return i;
419 };
420 Viskit.s.chart.jqplot.BarChart.prototype.getTicks = function(type){
421     var ticks = [];
422 
423     if(type === 'x'){
424         if(this.dataLabel().length == 0){
425             return this.xTicks();
426         }else{
427             var _dataField = this.traverseToDataField(this.data, this.dataField());
428 		
429             var dataGrpCount = 1;
430             if( _dataField instanceof Array ) {
431                 dataGrpCount = _dataField.length;
432 					
433             }
434             for(var j = 0 ; j <dataGrpCount; j++){
435                 ticks.push(this.getDataLabel(j));
436             };
437             return ticks.reverse();
438 				
439         }
440 			
441 			
442     }else if(type === 'y'){
443         return this.yTicks();
444     }
